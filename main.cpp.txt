
#include<iostream>
#include "ZipCode.h"
#include<string>

using namespace std;


int _tmain(int argc, _TCHAR* argv[])
{
	int zip;
	string str;
	int code;
	int n=0;
	int x=0;
	cout<<"please enter zipcode(zipcode integer):";
	cin>>zip;
	if(zip>99999){
		perror("input error");
		exit(1);
	}		
	ZipCode zipcodeint(zip);
	x=zipcodeint.check1();
	if(x==0){
		perror("input error");
		exit(1);
	}		
	zipcodeint.seperate();
	zipcodeint.encode();
	zipcodeint.display1();
	do{
	cout<<"please enter zipcode(zipcode bar code string):";
	cin>>str;
	}while(str.length()!=25);          //若輸入字串不為25digit則重新輸入
	
	ZipCode zipcodestr(str);
	zipcodestr.transform();
	
	n=zipcodestr.check2();
	if(n==0){
		perror("input error");
		exit(1);
	}
	zipcodestr.decode();
	zipcodestr.display2();
	system("pause");
	   


	return 0;
}
